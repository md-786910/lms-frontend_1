sudo nano /etc/nginx/sites-available/lms.conf

server {
    listen 80;
    server_name lms.leanport.com;

    # React build
    root /var/www/lms_build;
    index index.html;

    # SPA frontend
    location / {
        try_files $uri $uri/ /index.html;
    }

    # Backend API
    location /api/v1 {
        proxy_pass http://localhost:8000/api/v1;

        proxy_http_version 1.1;
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_set_header Upgrade           $http_upgrade;
        proxy_set_header Connection        "upgrade";
        proxy_cache_bypass                 $http_upgrade;
    }

    # Optional: static backend files (if served via backend)
    location /uploads {
        proxy_pass http://localhost:8000/uploads;

        proxy_http_version 1.1;
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # Socket.io support
    location /socket.io {
        proxy_pass http://localhost:8000/socket.io;

        proxy_http_version 1.1;
        proxy_set_header Upgrade           $http_upgrade;
        proxy_set_header Connection        "upgrade";
        proxy_set_header Host              $host;
        proxy_set_header X-Real-IP         $remote_addr;
        proxy_set_header X-Forwarded-For   $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;

        proxy_read_timeout 600s;
        proxy_send_timeout 600s;
    }

    # Optional: increase body size if you upload files
    # client_max_body_size 20M;
}


sudo ln -s /etc/nginx/sites-available/lms.conf /etc/nginx/sites-enabled/lms.conf

sudo certbot --nginx -d lms.leanport.com
